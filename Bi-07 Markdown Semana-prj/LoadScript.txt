DEPARTAMENTOS:
SELECT
	P.NO_DEPARTAMENTO
	,P.COD_DEPARTAMENTO
	,P.NO_SECAO
	,P.COD_SECAO
	,P.NO_GRUPO
	,P.COD_GRUPO
	,COD_PRODUTO
	,P.DESCRICAO AS NO_PRODUTO
	,P.COD_USUARIO
	,P.[COD_FORNECEDOR]
	,F.DESCRICAO AS NO_FORNECEDOR
	,P.CLASSIF_PRODUTO
FROM
	BI.DBO.BI_CAD_PRODUTO AS P LEFT JOIN BI_CAD_FORNECEDOR AS F ON (P.COD_FORNECEDOR = F.COD_FORNECEDOR)
WHERE 1=1;
