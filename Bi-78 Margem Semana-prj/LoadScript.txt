CMV:
SELECT 
	CMV.COD_LOJA
	,S.ANO_454
	,S.MES_454
	,S.NO_MES_454
	,S.SEMANA_454	
	,CONVERT(DATE,CMV.DTA_CALCULO) AS DATA
	,CMV.COD_PRODUTO		
	,CMV.MARGEM_REF	
	,SUM(CMV.Total_Qtde) as QTD_VENDA
	,SUM(CMV.Total_Venda) as VLR_VENDA
	,SUM(CMV.Total_Impostos) as VLR_IMPOSTOS
	,SUM(CMV.VLR_CUSTO_MEDIO*CMV.Total_Qtde) AS CMV		
FROM
	DW.[dbo].[CMV] as CMV
	INNER JOIN BI.DBO.BI_CAD_PRODUTO AS CP
		ON CMV.COD_PRODUTO = CP.COD_PRODUTO
	LEFT JOIN BI.dbo.BI_CAD_SEMANA AS S
		ON CMV.DTA_CALCULO = S.DATA
WHERE 1=1
	AND CONVERT(DATE,CMV.DTA_CALCULO) BETWEEN CONVERT(DATE,GETDATE()-90) AND CONVERT(DATE,GETDATE()-1)
	AND CMV.COD_PRODUTO NOT IN (1016980,1016982)
GROUP BY 	
	CMV.COD_LOJA
	,S.ANO_454
	,S.MES_454
	,S.NO_MES_454
	,S.SEMANA_454
	,CONVERT(DATE,CMV.DTA_CALCULO)
	,CMV.COD_PRODUTO		
	,CMV.MARGEM_REF;
